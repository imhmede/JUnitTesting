import java.text.DecimalFormat;

public class Test

{

	public static void main(String[] args) {
		
		//Round to 2 decimal places
		DecimalFormat dfTwo = new DecimalFormat("0.00");
		
		//Initial balance input
		Double dBalance = 1000.00;
		
		//Yearly interest, which is 5/100
	        Double dInterest = 0.05;
	    
	        //Initial year
		Integer year_count = 0;	
		
		//Total amount of years
		Integer total_year_count = 3;	
		
        while (year_count < total_year_count) {
        	
        	//Increment year until total year count is met
        	year_count++;
        	        	
        	//Add interest at end of year
        	dBalance = dBalance + (dBalance * dInterest);
        	
        	//Print the balance at end of year
        	System.out.println("The account balance will be $" + (dfTwo.format(dBalance)) + " after year " + year_count + ".");
        	   
        }
    }

}
